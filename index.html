// ============================================================ // メゾン富士 1階会議室 予約システム Code.gs v3 // ============================================================ var CONFIG = { spreadsheetName: 'メゾン富士_会議室予約', sheetName: '予約データ', adminEmail: '', }; var ALLOWED_ROOMS = [ '201','202','203','204','205', '301','302','303','304','305', '401','402','403','404','405', '501','502','503','504','505', '601' ]; var SLOTS = { morning: { label: '朝', time: '06:00〜12:00' }, afternoon: { label: '午後', time: '12:00〜18:00' }, evening: { label: '夜', time: '18:00〜22:00' } }; // ============================================================ // doGet: 全処理をGETパラメータで行う // callbackパラメータがあればJSONP形式で返す（CORS完全回避） // ============================================================ function doGet(e) { var action = e && e.parameter && e.parameter.action || ''; var callback = e && e.parameter && e.parameter.callback || ''; function respond(obj) { var json = JSON.stringify(obj); if (callback) { // JSONP形式：CORSを完全に回避できる return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } // アクションなし → HTMLアプリを返す if (!action) { return HtmlService .createHtmlOutputFromFile('index') .setTitle('メゾン富士 会議室予約') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } if (action === 'getReservations') { return respond(getReservations()); } if (action === 'addReservation') { var key = e.parameter.key || ''; var room = e.parameter.room || ''; var name = decodeURIComponent((e.parameter.name || '').replace(/\+/g, ' ')); return respond(addReservation(key, room, name)); } if (action === 'cancelReservation') { var key = e.parameter.key || ''; return respond(cancelReservation(key)); } return respond({ ok: false, error: '不明なアクション' }); } // ============================================================ // スプレッドシートを取得（なければ作成） // ============================================================ function getOrCreateSheet() { var files = DriveApp.getFilesByName(CONFIG.spreadsheetName); var ss; if (files.hasNext()) { ss = SpreadsheetApp.open(files.next()); } else { ss = SpreadsheetApp.create(CONFIG.spreadsheetName); var sheet = ss.getActiveSheet(); sheet.setName(CONFIG.sheetName); sheet.getRange(1, 1, 1, 7).setValues([[ '予約キー', '日付', '時間帯', '時間帯ラベル', '部屋番号', '氏名', '予約日時' ]]); sheet.setColumnWidth(1, 160); sheet.setColumnWidth(2, 100); sheet.setColumnWidth(3, 80); sheet.setColumnWidth(4, 80); sheet.setColumnWidth(5, 80); sheet.setColumnWidth(6, 120); sheet.setColumnWidth(7, 160); sheet.getRange(1, 1, 1, 7) .setBackground('#1a237e') .setFontColor('#ffffff') .setFontWeight('bold'); sheet.setFrozenRows(1); } var sheet = ss.getSheetByName(CONFIG.sheetName); if (!sheet) { sheet = ss.insertSheet(CONFIG.sheetName); sheet.getRange(1, 1, 1, 7).setValues([[ '予約キー', '日付', '時間帯', '時間帯ラベル', '部屋番号', '氏名', '予約日時' ]]); } return sheet; } // ============================================================ // 予約一覧取得 // ============================================================ function getReservations() { try { var sheet = getOrCreateSheet(); var lastRow = sheet.getLastRow(); var result = {}; if (lastRow >= 2) { var rows = sheet.getRange(2, 1, lastRow - 1, 7).getValues(); rows.forEach(function(row) { var resKey = row[0], room = row[4], name = row[5], ts = row[6]; if (resKey && room) { result[resKey] = { room: String(room), name: String(name), ts: String(ts) }; } }); } return { ok: true, data: result }; } catch (err) { return { ok: false, error: err.toString() }; } } // ============================================================ // 予約追加 // ============================================================ function addReservation(resKey, room, name) { try { name = name.trim(); if (!resKey || !room || !name) return { ok: false, error: '必須項目が不足しています' }; if (ALLOWED_ROOMS.indexOf(room) === -1) return { ok: false, error: '無効な部屋番号です' }; var parts = resKey.split('_'); var date = parts[0]; var slotId = parts[1]; if (!SLOTS[slotId]) return { ok: false, error: '無効な時間帯です' }; var sheet = getOrCreateSheet(); var lastRow = sheet.getLastRow(); // 二重予約チェック if (lastRow >= 2) { var keys = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); for (var i = 0; i < keys.length; i++) { if (keys[i][0] === resKey) return { ok: false, error: 'この時間帯はすでに予約されています' }; } } var now = Utilities.formatDate(new Date(), 'Asia/Tokyo', 'yyyy/MM/dd HH:mm:ss'); sheet.appendRow([resKey, date, slotId, SLOTS[slotId].label, room, name, now]); if (CONFIG.adminEmail) { try { GmailApp.sendEmail(CONFIG.adminEmail, '【予約通知】メゾン富士 会議室', '日付: '+date+'\n時間帯: '+SLOTS[slotId].label+'\n部屋: '+room+'号室\n氏名: '+name+'\n日時: '+now); } catch(e) { Logger.log('メールエラー: '+e); } } return { ok: true }; } catch (err) { return { ok: false, error: err.toString() }; } } // ============================================================ // 予約取消 // ============================================================ function cancelReservation(resKey) { try { if (!resKey) return { ok: false, error: 'キーが指定されていません' }; var sheet = getOrCreateSheet(); var lastRow = sheet.getLastRow(); if (lastRow < 2) return { ok: false, error: '予約データが見つかりません' }; var keys = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); var found = false; for (var i = keys.length - 1; i >= 0; i--) { if (keys[i][0] === resKey) { sheet.deleteRow(i + 2); found = true; break; } } if (!found) return { ok: false, error: '対象の予約が見つかりません' }; return { ok: true }; } catch (err) { return { ok: false, error: err.toString() }; } } // ============================================================ // セットアップ確認（手動実行用） // ============================================================ function setupCheck() { var sheet = getOrCreateSheet(); var ss = sheet.getParent(); Logger.log('✅ スプレッドシート: ' + ss.getName()); Logger.log('✅ URL: ' + ss.getUrl()); Logger.log('✅ セットアップ完了'); }
